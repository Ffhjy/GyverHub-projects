GHbutton b1, b2;

void build() {
  GHbutton b0;
  if (hub.Button(&b0)) {
    Serial.println(b0 ? "b0 press" : "b0 release");
  }
  hub.Button(&b1);
  hub.Button(&b2);
}

void loop() {
  if (b1) {
    static GHtimer tmr(500);
    if (tmr.ready()) Serial.println("hold b1");
  }

  if (b2.changed()) Serial.println(b2 ? "b2 press" : "b2 release");
}
